	/* ─── NAV SCROLL ─── */
	(function(){
		var nav = document.getElementById('pNav');
		window.addEventListener('scroll', function(){
			nav.classList.toggle('stuck', window.scrollY > 60);
		}, { passive: true });
	})();

	/* ─── SCROLL REVEAL (IntersectionObserver — same pattern as Framer Motion whileInView) ─── */
	(function(){
		var els = document.querySelectorAll('[data-reveal]');
		if (!('IntersectionObserver' in window)) {
			els.forEach(function(e){ e.classList.add('in'); });
			return;
		}
		var obs = new IntersectionObserver(function(entries){
			entries.forEach(function(en){
				if (en.isIntersecting) {
					en.target.classList.add('in');
					obs.unobserve(en.target);
				}
			});
		}, { threshold: 0.12 });
		els.forEach(function(e){ obs.observe(e); });
	})();

	/* ─── 3D CARD — ContainerScroll equivalent (rotateX on scroll) ─── */
	(function(){
		var card = document.getElementById('p3dCard');
		if (!card) return;
		var section = card.closest('.p-scroll-section');
		function onScroll(){
			var rect = section.getBoundingClientRect();
			var vh = window.innerHeight;
			/* progress 0→1 as section enters viewport */
			var progress = Math.min(1, Math.max(0, (vh - rect.top) / (vh + rect.height)));
			/* rotateX goes from 20deg → 0deg */
			var rx = 20 * (1 - progress);
			var sc = 1.04 - 0.04 * progress;
			card.style.transform = 'rotateX(' + rx + 'deg) scale(' + sc + ')';
		}
		window.addEventListener('scroll', onScroll, { passive: true });
		onScroll();
	})();

	/* ─── HERO 3D TILT (mouse parallax — Framer Motion useMotionValue equivalent) ─── */
	(function(){
		var card = document.getElementById('pTiltCard');
		if (!card) return;
		card.addEventListener('mousemove', function(e){
			var rect = card.getBoundingClientRect();
			var x = (e.clientX - rect.left) / rect.width - .5;
			var y = (e.clientY - rect.top) / rect.height - .5;
			card.style.transform = 'perspective(1000px) rotateY(' + (x * 12) + 'deg) rotateX(' + (-y * 8) + 'deg) scale(1.02)';
			card.style.transition = 'transform .1s ease';
		});
		card.addEventListener('mouseleave', function(){
			card.style.transform = 'perspective(1000px) rotateY(0) rotateX(0) scale(1)';
			card.style.transition = 'transform .5s cubic-bezier(0.22,1,0.36,1)';
		});
	})();
